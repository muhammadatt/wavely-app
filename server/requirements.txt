# Instant Polish — Python server dependencies

#
# torch and torchaudio are pinned to 2.0.0 because deepfilternet 0.5.6 uses

# ── PyTorch (pinned for deepfilternet compatibility) ─────────────────────────
torch==2.0.0

# deepfilter_enhance.py (Stage 2 noise reduction + NE-5 residual cleanup)

#   ~/.cache/DeepFilterNet/DeepFilterNet3

# ── Noise Eraser pipeline ─────────────────────────────────────────────────────

# rnnoise_denoise.py (NE-1 pre-separation pass)
pyrnnoise>=0.1.0

# separate_vocals.py (NE-3 — Demucs backend, default)
# htdemucs_ft model weights (~350 MB) downloaded on first run to:
#   ~/.cache/torch/hub/models--facebook--demucs/
demucs>=4.0.0

# separate_vocals.py (NE-3 — ConvTasNet backend, fast alternative)
# WHAM! checkpoint (~20 MB) downloaded from HuggingFace on first run.

# Diffusion model weights (~1.5 GB) downloaded on first run.

# Install after audiosr to avoid transformers version conflicts.
silero-vad>=5.1
